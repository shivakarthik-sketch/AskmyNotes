// backend/static/app.js document.addEventListener("DOMContentLoaded", function () { // Elements const pdfInput = document.getElementById("pdf-input"); const uploadStatus = document.getElementById("upload-status"); // File Input Behavior pdfInput.addEventListener("change", function () { const file = pdfInput.files[0]; // If user cancels selection or no file exists if (!file) { uploadStatus.textContent = ""; uploadStatus.className = ""; return; } // Show selected filename uploadStatus.textContent = `Selected "${file.name}" (ready to upload)`; // Apply success styling uploadStatus.className = "text-sm text-green-600 mt-2 min-h-[1.25rem]"; }); });
